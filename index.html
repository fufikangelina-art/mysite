import random import time import json from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup from telegram.ext import ApplicationBuilder, CommandHandler, CallbackQueryHandler, ContextTypes TOKEN = "ТУТ_ТВОЙ_TOKEN" ADMIN_ID = 123456789 DATA_FILE = "data.json" # 📂 завантаження def load_data(): try: with open(DATA_FILE, "r") as f: return json.load(f) except: return {} # 💾 збереження def save_data(data): with open(DATA_FILE, "w") as f: json.dump(data, f) users = load_data() def get_user(user_id): user_id = str(user_id) if user_id not in users: users[user_id] = { "energy": 10, "coins": 0, "last_update": time.time(), "last_click": 0 } return users[user_id] # 🔋 енергія def update_energy(user): now = time.time() diff = now - user["last_update"] regen = int(diff // 300) if regen > 0: user["energy"] = min(100, user["energy"] + regen) user["last_update"] = now # 📊 UI def get_text(user): progress = int((user["coins"] / 200) * 10) bar = "🟩" * progress + "⬜" * (10 - progress) return f""" 🦠 VIRUS TAP {bar} 🪙 {user['coins']} / 200 🔋 {user['energy']} / 100 """ def get_keyboard(): return InlineKeyboardMarkup([ [InlineKeyboardButton("🧪 ТАП", callback_data="tap")], [InlineKeyboardButton("📦 СУНДУК", callback_data="chest")], [InlineKeyboardButton("🎁 ЕНЕРГІЯ", callback_data="buy")] ]) # 🚀 старт async def start(update: Update, context: ContextTypes.DEFAULT_TYPE): user = get_user(update.effective_user.id) await update.message.reply_text(get_text(user), reply_markup=get_keyboard()) # 🧪 тап async def tap(update: Update, context: ContextTypes.DEFAULT_TYPE): query = update.callback_query user = get_user(query.from_user.id) update_energy(user) now = time.time() if now - user["last_click"] < 1: await query.answer("⏳ Повільніше") return user["last_click"] = now if user["energy"] > 0: user["energy"] -= 1 user["coins"] += 1 await query.answer("+1 🪙") else: await query.answer("❌ Немає енергії") save_data(users) await query.message.edit_text(get_text(user), reply_markup=get_keyboard()) # 📦 сундук async def chest(update: Update, context: ContextTypes.DEFAULT_TYPE): query = update.callback_query user = get_user(query.from_user.id) update_energy(user) if user["energy"] < 100: await query.answer("❌ Треба 100 енергії") return user["energy"] = 0 fake = " → ".join(str(random.choice([5,10,20,50])) for _ in range(5)) reward = random.choice([1,2,3,5]) user["coins"] += reward save_data(users) await query.message.reply_text(f""" 📦 ВІДКРИВАЄМО... 💰 {fake} 🎉 ВИПАЛО: {reward} 🪙 """) # 🎁 покупка async def buy(update: Update, context: ContextTypes.DEFAULT_TYPE): query = update.callback_query keyboard = [ [InlineKeyboardButton("✅ Я оплатив", callback_data="paid")] ] await query.message.reply_text( "🎁 Відправ подарунок і натисни кнопку", reply_markup=InlineKeyboardMarkup(keyboard) ) pending = {} # 📩 заявка async def paid(update: Update, context: ContextTypes.DEFAULT_TYPE): query = update.callback_query user = query.from_user pending[user.id] = True await context.bot.send_message( chat_id=ADMIN_ID, text=f"💰 Донат\n@{user.username}\nID: {user.id}" ) await query.message.reply_text("⏳ Чекаємо підтвердження") # ⚡ підтвердження async def confirm(update: Update, context: ContextTypes.DEFAULT_TYPE): user_id = str(context.args[0]) if user_id in users: users[user_id]["energy"] += 50 save_data(users) await context.bot.send_message( chat_id=int(user_id), text="✅ Оплата підтверджена\n⚡ +50 енергії" ) app = ApplicationBuilder().token(TOKEN).build() app.add_handler(CommandHandler("start", start)) app.add_handler(CommandHandler("confirm", confirm)) app.add_handler(CallbackQueryHandler(tap, pattern="tap")) app.add_handler(CallbackQueryHandler(chest, pattern="chest")) app.add_handler(CallbackQueryHandler(buy, pattern="buy")) app.add_handler(CallbackQueryHandler(paid, pattern="paid")) app.run_polling()
